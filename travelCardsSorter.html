Сортировщик карточек путешественника

Использовать так: 

    var cards = new TravelCadrs();
    var travelText = new Array();
 
    cards.add("Barcelona", "Gerona Airport", "the airport bus");
    cards.add("Madrid", "Barcelona", "train", "78A", "Seat 45B");
    cards.add("Stockholm", "New York JFK", "flight", "SK22", "Gate 22. Seat 7B. Baggage will be automatically transferred from your last leg");
    cards.add("Gerona Airport", "Stockholm", "flight", "SK455", "Gate 45B. Seat 3A. Baggage drop at ticket counter 344");
 
    cards.transportOrder({flight : 1});
 
    travelText=cards.takeCards_And_GiveMeTravel();
 
    var travel = document.getElementById('travel');
 
    for (var i = 0; i < travelText.length; i++) {
        var travelCardText = document.createElement('li');
        travelCardText.innerHTML = travelText[i];
        travel.appendChild(travelCardText);
    }
